MLEs

Poisson RV

Let \(X \sim \textrm{Poi}(\theta)\).

n <- 10000
lambda <- 5
x <- rpois(n, lambda)
hist(x)

Score RV

Define \(U = \frac{\partial}{\partial \theta} \log f(X | \theta)\).

U <- x / lambda - 1
hist(U)

Score variance

var(U)
## [1] 0.2000692
- mean(-x * lambda^(-2))
## [1] 0.198632

CLT for \(\bar{U}\)

it <- 1000
U_bar <- rep(NA, it)
for (i in 1:it) {
  n <- 500
  lambda <- 5
  x <- rpois(n, lambda)
  U <- x / lambda - 1
  U_bar[i] <- mean(U)
}

CLT for \(\bar{U}\)

hist(U_bar)

CLT for \(\bar{U}\)

var(U_bar)
## [1] 0.0003860197
- mean(-x * lambda^(-2)) / n
## [1] 0.0003984

CLT for \(\hat{\theta}^{MLE}\)

it <- 1000
MLE <- rep(NA, it)
for (i in 1:it) {
  n <- 500
  lambda <- 5
  x <- rpois(n, lambda)
  MLE[i] <- mean(x)
}

CLT for \(\hat{\theta}^{MLE}\)

hist(MLE)

CLT for \(\hat{\theta}^{MLE}\)

z <- sqrt(n * - mean(-x * lambda^(-2))) * (MLE - lambda)
hist(z)
